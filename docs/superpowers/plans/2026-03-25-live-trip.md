```typescript
import type { Itinerary, Day } from '../types/itinerary';

/**
 * Returns the current local date as YYYY-MM-DD in the given IANA timezone.
 * Uses Intl.DateTimeFormat with en-CA locale (natively emits YYYY-MM-DD).
 */
export function getTripLocalDate(timezone: string, now: Date = new Date()): string {
  return new Intl.DateTimeFormat('en-CA', {
    timeZone: timezone,
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',
  }).format(now);
}

/**
 * Classifies the trip as upcoming, live, or completed.
 * Uses string comparison on YYYY-MM-DD — safe because format is identical.
 */
export function getTripStatus(
  trip: Itinerary,
  now: Date = new Date(),
): 'upcoming' | 'live' | 'completed' {
  const localDate = getTripLocalDate(trip.timezone, now);
  if (localDate < trip.startDate) return 'upcoming';
  if (localDate > trip.endDate) return 'completed';
  return 'live';
}

export function isLiveTrip(trip: Itinerary, now: Date = new Date()): boolean {
  return getTripStatus(trip, now) === 'live';
}

/**
 * Returns the Day whose `date` matches the trip-local date, or undefined.
 */
export function getCurrentDay(trip: Itinerary, now: Date = new Date()): Day | undefined {
  const localDate = getTripLocalDate(trip.timezone, now);
  return trip.days.find(d => d.date === localDate);
}

export function getCurrentDayNumber(
  trip: Itinerary,
  now: Date = new Date(),
): number | undefined {
  return getCurrentDay(trip, now)?.dayNumber;
}

/**
 * Resolves `?day=` query param to concrete days + active filter label.
 *
 * Resolution rules (see spec for Rule D, E):
 *   null | ""           → { days: all, filter: 'all' }
 *   "today" live+match  → { days: [currentDay], filter: 'today' }
 *   "today" else        → { days: all, filter: 'all' }
 *   numeric N, exists   → { days: [day], filter: N }
 *   numeric N, missing  → { days: all, filter: 'all' }
 *   other string        → { days: all, filter: 'all' }
 */
export function getVisibleDays(
  trip: Itinerary,
  dayParam: string | null,
  now: Date = new Date(),
): { days: Day[]; filter: 'all' | 'today' | number } {
  if (!dayParam) return { days: trip.days, filter: 'all' };

  if (dayParam === 'today') {
    const currentDay = isLiveTrip(trip, now) ? getCurrentDay(trip, now) : undefined;
    if (!currentDay) return { days: trip.days, filter: 'all' };
    return { days: [currentDay], filter: 'today' };
  }

  const n = parseInt(dayParam, 10);
  if (isNaN(n)) return { days: trip.days, filter: 'all' };

  const matched = trip.days.filter(d => d.dayNumber === n);
  if (matched.length === 0) return { days: trip.days, filter: 'all' };

  return { days: matched, filter: n };
}
```

import TripLayout from '../../layouts/TripLayout.astro';
import TripHeader from '../../components/TripHeader.astro';
import DayFilter from '../../components/DayFilter.astro';
import DaySection from '../../components/DaySection.astro';
import TripMap from '../../components/TripMap.astro';
import PinGateForm from '../../components/PinGateForm.astro';
import LiveTripBanner from '../../components/LiveTripBanner.astro';
import { env } from 'cloudflare:workers';
import { getTrip } from '../../lib/kv';
import { verifyTripCookie, cookieName, parseCookies } from '../../lib/cookie';
import {
  getTripStatus,
  getCurrentDay,
  getCurrentDayNumber,
  isLiveTrip,
  getVisibleDays,
} from '../../lib/trip-state';

const { id } = Astro.params as { id: string };

// 1. Load trip
const trip = await getTrip(env.TRIPS, id);
if (!trip) {
  return new Response(null, { status: 404 });
}

// 2. Check access cookie
const cookies = parseCookies(Astro.request.headers.get('Cookie'));
const rawCookie = cookies[cookieName()];
let authorized = false;
if (rawCookie) {
  const tripId = await verifyTripCookie(rawCookie, env.COOKIE_SIGNING_SECRET);
  authorized = tripId === id;
}

// 3. Compute trip state
const now = new Date();
const dayParam = Astro.url.searchParams.get('day');
const { days: visibleDays, filter: activeFilter } = getVisibleDays(trip, dayParam, now);
const tripStatus = getTripStatus(trip, now);
const isLive = tripStatus === 'live';
const currentDay = getCurrentDay(trip, now);
const currentDayNumber = currentDay?.dayNumber;

// 4. Map items scoped to visible days
const mapItems = visibleDays.flatMap(d => d.items);

// 5. Prev/next day links (only when viewing a single numeric day)
const singleDay = typeof activeFilter === 'number' ? activeFilter : null;
const prevDayNumber = singleDay !== null
  ? trip.days.find(d => d.dayNumber === singleDay - 1)?.dayNumber
  : undefined;
const nextDayNumber = singleDay !== null
  ? trip.days.find(d => d.dayNumber === singleDay + 1)?.dayNumber
  : undefined;

// 6. Formatted local date for live banner (e.g. "Wed, Mar 25")
const localDateDisplay = isLive
  ? new Intl.DateTimeFormat('en-US', {
      timeZone: trip.timezone,
      weekday: 'short',
      month: 'short',
      day: 'numeric',
    }).format(now)
  : '';
---
<TripLayout title={trip.title}>
  {!authorized ? (
    <PinGateForm tripId={id} />
  ) : (
    <main>
      {isLive && currentDayNumber !== undefined && (
        <LiveTripBanner
          currentDayNumber={currentDayNumber}
          totalDays={trip.days.length}
          localDate={localDateDisplay}
          tripId={id}
        />
      )}
      <TripHeader trip={trip} />
      <TripMap
        items={mapItems}
        centerLat={activeFilter === 'all' ? trip.map?.centerLat : undefined}
        centerLng={activeFilter === 'all' ? trip.map?.centerLng : undefined}
        zoom={activeFilter === 'all' ? trip.map?.zoom : undefined}
      />
      <DayFilter
        days={trip.days}
        activeFilter={activeFilter}
        tripId={id}
        isLive={isLive}
        currentDayNumber={currentDayNumber}
      />
      <div class="py-4">
        {visibleDays.length === 0 ? (
          <p class="px-4 py-8 text-center text-sm text-stone-400">
            No days to show. <a href={`/trip/${id}`} class="underline">View all days</a>
          </p>
        ) : (
          visibleDays.map(day => (
            <DaySection
              day={day}
              isToday={day.dayNumber === currentDayNumber}
            />
          ))
        )}
      </div>

      {/* Prev/next navigation — only when viewing a single numeric day */}
      {singleDay !== null && (prevDayNumber !== undefined || nextDayNumber !== undefined) && (
        <nav
          aria-label="Day navigation"
          class="flex items-center justify-between px-4 pb-8 pt-2 text-sm sm:px-6"
        >
          <div>
            {prevDayNumber !== undefined ? (
              <a
                href={`/trip/${id}?day=${prevDayNumber}`}
                class="text-stone-500 hover:text-stone-900 focus:outline-none focus-visible:ring-2 focus-visible:ring-stone-400 rounded"
              >
                ← Day {prevDayNumber}
              </a>
            ) : <span />}
          </div>
          <div>
            {nextDayNumber !== undefined ? (
              <a
                href={`/trip/${id}?day=${nextDayNumber}`}
                class="text-stone-500 hover:text-stone-900 focus:outline-none focus-visible:ring-2 focus-visible:ring-stone-400 rounded"
              >
                Day {nextDayNumber} →
              </a>
            ) : <span />}
          </div>
        </nav>
      )}
    </main>
  )}
</TripLayout>
