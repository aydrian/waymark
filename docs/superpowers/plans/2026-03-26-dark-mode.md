# Dark Mode Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add light/dark/system theme switching to the Waymark Astro app, persisted in `localStorage`, with a floating popover toggle using Lucide icons.

**Architecture:** CSS custom properties define the full color palette in `global.css`; a `html.dark` override swaps them. A blocking inline `<script>` in `<head>` applies the `dark` class before first paint (no FOUC). A new `ThemeToggle.astro` floating button lets users pick System / Light / Dark via a popover. All 9 existing components swap hardcoded Tailwind color classes for `[var(--color-*)]` arbitrary values.

**Tech Stack:** Astro 6, Tailwind CSS v4 (`@tailwindcss/vite`), `lucide-astro`, Bun

---

```astro
---

<div id="theme-wrapper" class="fixed bottom-6 right-6 z-50">
  <!-- Popover -->
  <div

    aria-label="Theme options"
    class="hidden absolute bottom-14 right-0 w-36 rounded-xl border border-[var(--color-border)] bg-[var(--color-surface)] shadow-lg overflow-hidden"

      role="menuitem"
      data-theme="system"
      class="flex w-full items-center gap-2.5 px-4 py-2.5 text-sm text-[var(--color-text-secondary)] hover:bg-[var(--color-bg-subtle)] transition-colors"
    >
      <Monitor size={15} aria-hidden="true" />
      System
    </button>
    <button
      role="menuitem"
      data-theme="light"
      class="flex w-full items-center gap-2.5 px-4 py-2.5 text-sm text-[var(--color-text-secondary)] hover:bg-[var(--color-bg-subtle)] transition-colors"
    >
      <Sun size={15} aria-hidden="true" />
      Light
    </button>
    <button
      role="menuitem"
      data-theme="dark"
      class="flex w-full items-center gap-2.5 px-4 py-2.5 text-sm text-[var(--color-text-secondary)] hover:bg-[var(--color-bg-subtle)] transition-colors"
    >
      <Moon size={15} aria-hidden="true" />
      Dark
    </button>
  </div>

  <!-- Toggle button -->
  <button
    id="theme-btn"
    aria-label="Change theme"
    aria-haspopup="true"
    aria-expanded="false"
    aria-controls="theme-popover"
    class="flex h-10 w-10 items-center justify-center rounded-full border border-[var(--color-border)] bg-[var(--color-surface)] text-[var(--color-text-secondary)] shadow-md hover:bg-[var(--color-bg-subtle)] transition-colors focus:outline-none focus-visible:ring-2 focus-visible:ring-[var(--color-focus-ring)]"
  >
    <span id="icon-system" class="hidden"><Monitor size={18} aria-hidden="true" /></span>
    <span id="icon-light"  class="hidden"><Sun    size={18} aria-hidden="true" /></span>
    <span id="icon-dark"   class="hidden"><Moon   size={18} aria-hidden="true" /></span>
  </button>
</div>

<script>
  const STORAGE_KEY = 'waymark-theme';
  type Theme = 'system' | 'light' | 'dark';

  function getStored(): Theme {
    const v = localStorage.getItem(STORAGE_KEY);
    if (v === 'light' || v === 'dark' || v === 'system') return v;
    return 'system';
  }

  function prefersDark(): boolean {
    return window.matchMedia('(prefers-color-scheme: dark)').matches;
  }

  function applyTheme(theme: Theme) {
    const isDark = theme === 'dark' || (theme === 'system' && prefersDark());
    document.documentElement.classList.toggle('dark', isDark);
  }

  function showIcon(theme: Theme) {
    ['system', 'light', 'dark'].forEach(t => {
      const el = document.getElementById(`icon-${t}`);
      if (el) el.classList.toggle('hidden', t !== theme);
    });
  }

  function highlightActive(theme: Theme) {
    // Use inline styles to avoid Tailwind purging dynamically-added arbitrary classes
    document.querySelectorAll<HTMLButtonElement>('[data-theme]').forEach(btn => {
      const isActive = btn.dataset.theme === theme;
      btn.style.fontWeight = isActive ? '600' : '';
      btn.style.color = isActive ? 'var(--color-text-primary)' : '';
    });
  }

  // ── Init ──────────────────────────────────────────────────────
  let current = getStored();
  showIcon(current);
  highlightActive(current);

  // ── System preference listener ────────────────────────────────
  let mql: MediaQueryList | null = null;
  function onSystemChange() { applyTheme('system'); }

  function updateSystemListener(theme: Theme) {
    if (!mql) mql = window.matchMedia('(prefers-color-scheme: dark)');
    if (theme === 'system') {
      mql.addEventListener('change', onSystemChange);
    } else {
      mql.removeEventListener('change', onSystemChange);
    }
  }
  updateSystemListener(current);

  // ── Popover toggle ────────────────────────────────────────────
  const btn = document.getElementById('theme-btn')!;
  const popover = document.getElementById('theme-popover')!;

  btn.addEventListener('click', (e) => {
    e.stopPropagation();
    const open = popover.classList.toggle('hidden') === false;
    btn.setAttribute('aria-expanded', String(open));
  });

  document.addEventListener('click', () => {
    if (!popover.classList.contains('hidden')) {
      popover.classList.add('hidden');
      btn.setAttribute('aria-expanded', 'false');
    }
  });

  // ── Option selection ──────────────────────────────────────────
  popover.querySelectorAll<HTMLButtonElement>('[data-theme]').forEach(optBtn => {
    optBtn.addEventListener('click', (e) => {
      e.stopPropagation();
      const theme = optBtn.dataset.theme as Theme;
      localStorage.setItem(STORAGE_KEY, theme);
      current = theme;
      applyTheme(theme);
      showIcon(theme);
      highlightActive(theme);
      updateSystemListener(theme);
      popover.classList.add('hidden');
      btn.setAttribute('aria-expanded', 'false');
    });
  });
</script>
```

```astro
import '../styles/global.css';
import ThemeToggle from '../components/ThemeToggle.astro';
interface Props {
  title: string;
}
const { title } = Astro.props;
---
<!doctype html>
<html lang="en" class="h-full">
  <head>
    <meta charset="utf-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1" />
    <meta name="robots" content="noindex, nofollow" />
    <link rel="icon" type="image/svg+xml" href="/favicon.svg" />
    <title>{title} — Waymark</title>
    <!-- Blocking theme script: runs before paint to prevent FOUC -->
    <script is:inline>
      (function () {
        var stored = localStorage.getItem('waymark-theme');
        var prefersDark = window.matchMedia('(prefers-color-scheme: dark)').matches;
        var isDark = stored === 'dark' || (stored !== 'light' && prefersDark);
        if (isDark) document.documentElement.classList.add('dark');
      })();
    </script>
  </head>
  <body class="min-h-full bg-[var(--color-bg)] text-[var(--color-text-primary)] antialiased">
    <slot />
    <ThemeToggle />
  </body>
</html>
```

> Note: Astro requires `is:inline` on scripts that must run synchronously and unprocessed. Without it, Astro defers/bundles the script, which would cause FOUC.

`src/components/TripMap.astro`:
```astro
---
import type { TripItem } from '../types/itinerary';
import 'leaflet/dist/leaflet.css';

interface Props {
  items: TripItem[];
  centerLat?: number;
  centerLng?: number;
  zoom?: number;
}
const { items, centerLat, centerLng, zoom } = Astro.props;
const coordItems = items.filter(i => i.lat !== undefined && i.lng !== undefined);

const mapData = JSON.stringify({
  items: coordItems.map(i => ({
    lat: i.lat,
    lng: i.lng,
    title: i.title,
    type: i.type,
    location: i.location,
    address: i.address,
  })),
  center: centerLat !== undefined && centerLng !== undefined ? [centerLat, centerLng] : null,
  zoom: zoom ?? null,
});
---
{coordItems.length > 0 && (
  <div class="px-4 pt-4 sm:px-6">
    <div
      id="trip-map"
      class="rounded-xl overflow-hidden border border-[var(--color-border)] h-52"
      data-map={mapData}
    ></div>
  </div>
)}

<script>
  import L from 'leaflet';

  const el = document.getElementById('trip-map') as HTMLElement | null;
  if (el) {
    const { items, center, zoom } = JSON.parse(el.dataset.map!);
    const isDark = document.documentElement.classList.contains('dark');
    const markerBorder = isDark ? '#44403c' : '#ffffff';

    const map = L.map(el, { zoomControl: true, scrollWheelZoom: false });

    L.tileLayer('https://tile.openstreetmap.org/{z}/{x}/{y}.png', {
      attribution: '© <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors',
      maxZoom: 19,
    }).addTo(map);

    const TYPE_COLORS: Record<string, string> = {
      hotel: '#3b82f6',
      restaurant: '#f97316',
      activity: '#22c55e',
      transport: '#6b7280',
      transfer: '#6b7280',
      note: '#eab308',
    };

    const markers: L.Marker[] = [];
    for (const item of items) {
      const color = TYPE_COLORS[item.type] ?? '#6b7280';
      const icon = L.divIcon({
        className: '',
        html: `<div style="width:12px;height:12px;border-radius:50%;background:${color};border:2px solid ${markerBorder};box-shadow:0 1px 3px rgba(0,0,0,.4)"></div>`,
        iconSize: [12, 12],
        iconAnchor: [6, 6],
      });
      const marker = L.marker([item.lat, item.lng], { icon });
      const detail = item.address ?? item.location;
      const popup = `<strong>${item.title}</strong>${detail ? `<br><span style="font-size:0.75rem">${detail}</span>` : ''}`;
      marker.bindPopup(popup);
      marker.addTo(map);
      markers.push(marker);
    }

    if (center) {
      map.setView(center, zoom ?? 12);
    } else if (markers.length > 0) {
      const group = L.featureGroup(markers);
      map.fitBounds(group.getBounds().pad(0.2));
    }
  }
</script>
```
